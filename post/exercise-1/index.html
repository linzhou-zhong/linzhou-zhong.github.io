(define (square x)(* x x))

(define (sum_square x y) (+ (square x) (square y)))

(define (sqsumlargest x y z)
    (cond 
        ((and (>= y x) (>= z x)) (sum_square y z))
        ((and (>= x y) (>= z y)) (sum_square x z))
        ((and (>= x z) (>= y z)) (sum_square x y))
    )
)
